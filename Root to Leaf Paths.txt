/*
Definition for Node
class Node
{
    int data;
    Node left;
    Node right;

    Node(int val)
    {
        this.data = val;
        left = null;
        right = null;
    }
}
*/

class Solution {
    public ArrayList<ArrayList<Integer>> Paths(Node root) {
        // code here
        ArrayList<ArrayList<Integer>> result = new ArrayList<>();
        if(root==null)
        {
            return result;
        }
        getPath(root, new ArrayList<>(),result);
        return result;
        
    }
    
    public void getPath(Node node , ArrayList<Integer>path, ArrayList<ArrayList<Integer>> result)
    {
        if(node ==null)
        {
            return ;
        }
        path.add(node.data);
        if(node.left==null && node.right==null)
        {
            result.add(new ArrayList<>(path));
        }else{
            getPath(node.left,path,result);
            getPath(node.right,path,result);
        }
        path.remove(path.size()-1);
    }
}
