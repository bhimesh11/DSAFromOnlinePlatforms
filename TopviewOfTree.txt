class pair {
	int hd;
	Node node;
	
	pair(Node node, int hd)
	{
		this.hd = hd;
		this.node = node;
	}
}
class Solution {
	public ArrayList<Integer> topView(Node root) {
		ArrayList<Integer> ans = new ArrayList<Integer>();
		if (root == null)
			{
			return ans;
		}
		Map<Integer, Integer> map = new TreeMap<>();
		Queue<pair> pairQueue = new LinkedList<>();
		pairQueue.add(new pair(root, 0));
		while (!pairQueue.isEmpty())
			{
			pair it = pairQueue.remove();
			int val = it.hd;
			Node temp = it.node;
			
			if (!map.containsKey(val)) {
				map.put(val, temp.data);
			}
			
			if (temp.left != null)
				{
				pairQueue.add(new pair(temp.left, val - 1));
			}
			if (temp.right != null)
				{
				pairQueue.add(new pair(temp.right, val + 1));
			}
			
		}
		for (Map.Entry<Integer, Integer> m: ())
			{
			ans.add(m.getValue());
		}
		return ans;
	}
}
